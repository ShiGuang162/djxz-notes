### 9.3 错误处理最佳实践

```javascript
loadData() {
  if (this.data.isLoading) return
  this.setData({ isLoading: true })

  try {
    const data = fetchData()
    this.setData({ data, isLoading: false })
  } catch (error) {
    console.error('加载失败:', error)
    this.setData({ 
      data: [], // 设置默认值
      isLoading: false 
    })
    wx.showToast({ title: '加载失败', icon: 'none' })
  }
}
```
